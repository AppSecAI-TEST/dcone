create table dc_user(uid int primary key auto_increment,itcode varchar(16) unique not null,username varchar(32)not null,islocked int,isOnthescene int);
#dc_user的表
create table dc_wallet(wid int primary key auto_increment,uid int not null references dc_user(uid),amount int);
#用户钱包的表
create table dc_trade(tid int primary key auto_increment,wid int not null references dc_wallet(wid),volume int not null,tradetime datetime,memo varchar(50));
#交易记录的表
create view dc_user_wallet as select a.uid as uid, a.itcode as itcode,a.username as username,b.amount as amount from dc_user as a, dc_wallet as b where a.uid=b.uid;
#用户账户信息的视图
create table lucky_money (l_id int auto_increment primary key,round int unique not null,total int not null);
#红包的表
create table lucky_money_record(r_id int auto_increment primary key ,wid int not null,luck_number int not null,round int not null);
